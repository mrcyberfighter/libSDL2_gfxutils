Check the validity of the SDL_Renderer.

if (pRenderer == NULL) {
  fprintf(stderr,"SDL Renderer error (%s)\n",SDL_GetError());
  exit(EXIT_FAILURE) ;
}

void

Check only if the given parameter is equal to NULL.

if (form == NULL) {
  fprintf(stderr,"Invalid form argument !\n");
  exit(EXIT_FAILURE) ;
}

Note
